;~~~~~~~~~~~~~~~~~~~~~;
; Recursive factorial ;    
;~~~~~~~~~~~~~~~~~~~~~;

    in

    pop     ax
    push    ax

    call    factorial
    out
    ret

factorial:

    push ax
    push 0
    je  zero

    

zero:

int fact (int num)
{
    if (num == 0)
        return 1;
    else
        return num * fact (num - 1);
}
